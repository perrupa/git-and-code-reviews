# Aliases

```bash
[alias]
	# Shortcuts
	a        = add
	aa       = add --all
	co       = checkout
	ci       = commit
	cia      = commit --amend --no-edit
	st       = status
	pr       = pull --rebase
	type     = cat-file -t
	dump     = cat-file -p
	lg       = log --all --graph --pretty=oneline --decorate

	# Normalize listing 
	branches = branch -a
	remotes  = remote -v
	tags     = tag -l```
